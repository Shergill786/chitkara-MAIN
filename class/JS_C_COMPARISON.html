⚡ JavaScript vs C Comparison ⚡

Side-by-Side Reference Guide for Functions, Tools & Statements

1️⃣ Basic Output & Input

JavaScript
console.log("Hello, World!"); console.log("Age:", age); console.log(a + b);

Uses console.log() to output to console. Automatically converts types to strings.

C
#include <stdio.h> printf("Hello, World!\n"); printf("Age: %d\n", age); printf("%d\n", a + b);

Uses printf() for formatted output. Requires format specifiers (%d, %f, %s, etc.)

JavaScript
let email = prompt("Enter email:"); let num = Number(prompt("Enter number:"));

Uses prompt() for user input. Returns string by default.

C
int num; char email[50]; scanf("%d", &num); scanf("%s", email);

Uses scanf() for input. Requires address operator (&) for variables.

2️⃣ Variables & Data Types

JavaScript
let age = 24; const pi = 3.14; var score = 100; let x = null; let y = undefined; let isActive = true; let name = "JavaScript";

Dynamic typing. let and const are block-scoped. var is function-scoped.

C
#include <stdio.h> int age = 24; float pi = 3.14; char name[] = "C Language"; int x = 0; float y = 3.14; char isActive = 'Y';

Static typing. Must declare type before use. Common types: int, float, char, double

Operation JavaScript C

let a = 10, b = 5; console.log(a + b); // 15 console.log(a - b); // 5 console.log(a * b); // 50 console.log(a / b); // 2 console.log(a % b); // 0 console.log(a ** 2); // 100

#include <stdio.h> int a = 10, b = 5; printf("%d\n", a + b); // 15 printf("%d\n", a - b); // 5 printf("%d\n", a * b); // 50 printf("%d\n", a / b); // 2 printf("%d\n", a % b); // 0

4️⃣ Comparison Operators

Operation JavaScript C
Equal to a == b, a === b a == b
Not equal a != b, a !== b a != b
Greater than a > b a > b
Less than a < b a < b
Greater or equal a >= b a >= b
Less or equal a <= b a <= b
JavaScript
let a = 10, b = 5; console.log(a > b); // true console.log(a < b); // false console.log(a == b); // false console.log(a != b); // true console.log(a >= 10); // true console.log(a <= 5); // false
C
int a = 10, b = 5; printf("%d\n", a > b); // 1 (true) printf("%d\n", a < b); // 0 (false) printf("%d\n", a == b); // 0 (false) printf("%d\n", a != b); // 1 (true) printf("%d\n", a >= 10); // 1 (true) printf("%d\n", a <= 5); // 0 (false)

5️⃣ Logical Operators

JavaScript
let a = 10, b = 5; // AND operator console.log(a > 5 && b < 10); // true // OR operator console.log(a > 5 || b < 10); // true // NOT operator console.log(!(a < b)); // true

Logical AND (&&), OR (||), NOT (!)

C
int a = 10, b = 5; // AND operator printf("%d\n", (a > 5 && b < 10)); // 1 // OR operator printf("%d\n", (a > 5 || b < 10)); // 1 // NOT operator printf("%d\n", !(a < b)); // 1

Same operators: && (AND), || (OR), ! (NOT). Returns 1 (true) or 0 (false)

6️⃣ Conditional Statements (if/else)

JavaScript
let num = 85; if (num >= 90) { console.log("Grade A"); } else if (num >= 80) { console.log("Grade B"); } else if (num >= 70) { console.log("Grade C"); } else { console.log("Grade F"); }
C
#include <stdio.h> int num = 85; if (num >= 90) { printf("Grade A\n"); } else if (num >= 80) { printf("Grade B\n"); } else if (num >= 70) { printf("Grade C\n"); } else { printf("Grade F\n"); }

7️⃣ Switch Statement

JavaScript
let choice = 2; let result; switch (choice) { case 1: result = "Addition"; break; case 2: result = "Subtraction"; break; case 3: result = "Multiplication"; break; default: result = "Invalid"; }
C
#include <stdio.h> int choice = 2; switch (choice) { case 1: printf("Addition\n"); break; case 2: printf("Subtraction\n"); break; case 3: printf("Multiplication\n"); break; default: printf("Invalid\n"); }
⚠️ Important: Both languages require break; statements to prevent fall-through between cases.

9️⃣ Loops - WHILE

JavaScript

#include <stdio.h> int count = 1; while (count <= 5) { printf("Count: %d\n", count); count++; } // Output: // Count: 1 // Count: 2 // Count: 3 // Count: 4 // Count: 5

1️⃣5️⃣.5️⃣ Recursion

JavaScript
// Recursive function - Factorial function factorial(n) { if (n <= 1) { return 1; } return n * factorial(n - 1); } console.log(factorial(5)); // 120 // Recursive function - Fibonacci function fibonacci(n) { if (n <= 1) { return n; } return fibonacci(n - 1) + fibonacci(n - 2); } console.log(fibonacci(6)); // 8

Recursion is when a function calls itself. Must have a base case to avoid infinite loops.

C
#include <stdio.h> // Recursive function - Factorial int factorial(int n) { if (n <= 1) { return 1; } return n * factorial(n - 1); } // Recursive function - Fibonacci int fibonacci(int n) { if (n <= 1) { return n; } return fibonacci(n - 1) + fibonacci(n - 2); } int main() { printf("%d\n", factorial(5)); // 120 printf("%d\n", fibonacci(6)); // 8 return 0; }

Recursion works the same way in C. Function calls itself with a base case to stop recursion.

📌 JavaScript Note: JavaScript doesn't have pointers like C. It has automatic memory management.

#include <stdio.h> int a = 10; // Pointer declaration int* ptr = &a; // ptr points to a // Access value printf("%d\n", *ptr); // 10 // Access address printf("%p\n", ptr); // Memory address // Pointer to pointer int** ptr2 = &ptr; printf("%d\n", **ptr2); // 10 // Array as pointer int arr[5] = {1,2,3,4,5}; int* p = arr; printf("%d\n", p[0]); // 1

2️⃣0️⃣ Dynamic Memory Allocation (C)

📌 JavaScript Note: Memory management is automatic. No need for manual allocation/deallocation.
C
#include <stdio.h> #include <stdlib.h> // Allocate memory int* ptr = (int*)malloc(sizeof(int) * 5); // Use memory ptr[0] = 10; ptr[1] = 20; // Free memory free(ptr); ptr = NULL; // Good practice // Using calloc (initializes to 0) int* arr = (int*)calloc(10, sizeof(int)); // Reallocate memory ptr = (int*)realloc(ptr, sizeof(int) * 10); free(arr);

malloc() = allocate, free() = deallocate, calloc() = allocate and initialize

2️⃣1️⃣ Number to Array Conversion

JavaScript
// Method 1: Using string let num = 12345; let digits = num .toString() .split('') .map(Number); console.log(digits); // [1, 2, 3, 4, 5] // Method 2: Manual extraction let num2 = 54321; let arr = []; while (num2 > 0) { arr.push(num2 % 10); num2 = Math.floor(num2 / 10); } arr.reverse(); console.log(arr); // [5, 4, 3, 2, 1]
C
#include <stdio.h> // Method: Manual extraction int num = 12345; int digits[10]; int count = 0; int temp = num; while (temp > 0) { digits[count] = temp % 10; count++; temp /= 10; } // Reverse for proper order printf("Digits: "); for (int i = count - 1; i >= 0; i--) { printf("%d ", digits[i]); } // Digits: 1 2 3 4 5

2️⃣2️⃣ Deep Copy vs Shallow Copy

JavaScript
// Shallow copy (shares nested objects) let obj1 = {a: 1, b: {c: 2}}; let obj2 = {...obj1}; obj2.b.c = 3; console.log(obj1.b.c); // 3 (changed!) // Deep copy (independent copy) let obj3 = {a: 1, b: {c: 2}}; let obj4 = JSON.parse( JSON.stringify(obj3) ); obj4.b.c = 3; console.log(obj3.b.c); // 2 (unchanged)
C
#include <stdio.h> #include <string.h> // Shallow copy (shares memory) int arr1[3] = {1, 2, 3}; int* ptr1 = arr1; int* ptr2 = ptr1; // Deep copy (separate memory) int arr2[3] = {1, 2, 3}; int arr3[3]; memcpy(arr3, arr2, sizeof(arr2)); // Or manual copy for (int i = 0; i < 3; i++) { arr3[i] = arr2[i]; }
