C
#include // Recursive factorial int factorial(int n) { // Base case if (n <= 1) { return 1; } // Recursive case return n * factorial(n - 1); } int main() { printf("%d\n", factorial(5)); // 120 printf("%d\n", factorial(0)); // 1 // How it works: // factorial(5) = 5 * factorial(4) // factorial(4) = 4 * factorial(3) // ... until factorial(1) = 1 return 0; }

#include // Fibonacci: 0, 1, 1, 2, 3, 5, 8... int fibonacci(int n) { if (n <= 1) { return n; } return fibonacci(n - 1) + fibonacci(n - 2); } int main() { printf("%d\n", fibonacci(6)); // 8 printf("%d\n", fibonacci(7)); // 13 // Print first 10 Fibonacci numbers for (int i = 0; i < 10; i++) { printf("%d\n", fibonacci(i)); } return 0; }
